static VALUE
rupee_corr(self, _xvals, _yvals)
  VALUE self, _xvals, _yvals;
{
  int len = RARRAY_LEN(_xvals);
  double xvals[len], yvals[len];

  rtofa(xvals, _xvals, len);
  rtofa(yvals, _yvals, len);

  return rb_float_new(corr(xvals, yvals, len));
}
