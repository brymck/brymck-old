def black_scholes(call_put_flag, s, x, t, r, v)
  d1 = (Math.log(s / x) + (r + v * v / 2.0) * t) / (v * Math.sqrt(t))
  d2 = d1 - v * Math.sqrt(t)
  if call_put_flag == 'c'
    s * cnd(d1) - x * Math.exp(-r * t) * cnd(d2)
  else
    x * Math.exp(-r * t) * cnd(-d2) - s * cnd(-d1)
  end
end
